Modulo http - cria um servidor web configurado na porta 8081 localhost:8081

var http = require('http');
http.createServer(function (request, response) {
 response.end('Hello World');
}).listen(8081);

Modulo fs - acesso ao sistema de arquivo do SO (file system)

var a = require('fs');
console.log(__dirname+'\\doc.txt');
var texto = a.readFile(__dirname+'/doc.txt', 
 function(erro, retorno){
  if(erro)
   console.log('erro de acesso');
  else
   console.log('r:'+retorno);
 });
